栈

栈是一种数据结构
先进后出后进先出
push入栈 pop出栈

{
    #include "stack.h"
// 栈空间(结构) : 先进后出，后进先出

// 初始化栈空间
void stack_init( stack_t* s, int cap ) {
    s->arr = malloc(sizeof(int)*cap);
    s->cap = cap;
    s->top = 0;
}
// 释放栈空间
void stack_deinit( stack_t* s ) {
    free(s->arr);
    s->cap = 0;
    s->top = 0;
}
// 判断栈空间是否已满
int stack_full( stack_t* s ) {
    return s->top == s->cap; // 满返回1，否则返回0
}
// 判断栈空间是否已空
int stack_empty( stack_t* s ) {
    return s->top == 0; // 空返回1，否则返回0
}
// 压栈（存入数据）
void stack_push( stack_t* s, int data ) {
    s->arr[s->top++] = data;
}//先用后加
// 弹栈(取出数据)
int stack_pop( stack_t* s ) {
   return s->arr[--s->top];
}
// 获取栈空间目前存储数据的个数
int stack_size( stack_t* s ) {
    return s->top;
}
}

队列

另一种存储空间
栈的要求：先进后出…… 入栈出栈 压栈弹栈
队列要求：先进先出……FIFO 入队出队

{
    typedef struct queue{
        int *arr;//指向队列结构首地址（队列空间）
        int cap;//容量
        int rear;//队列的尾
        int front;//队列的头
        int size; //已经存了多少个
    } queue_t;

    //初始化
    void queue_init(queue_t* pq, int cap){
        pq->arr = malloc(sizeof(int )*cap);
        pq->cap = cap;
        pq->size=0;
        pq->front=pq->rear=0;
    }

    //释放队列
    void queue_deinit(queue_t *pq){
        free(pq->arr);
        pq->cap=pq->size=pq->front=pq->rear=0;
    }

    //队列判满
    int queue_full(queue_t *pq){
        return pq->size==pq->cap;
    }

    //队列判空
    int queue_empty(queue_t* pq){
        retuan pq->size==0;
    }

    //入队
    void queue_push (queue_t * pq,int data){
        if(pq->rear>=pq->cap && !queue_empty(pq)){
            pq->rear=0;
        }
        pq->arr[pq->rear++]=data;
        pq->size++;
    }

    //出队
    int queue_pop( queue_t* pq ) {
    if( pq->front>=pq->cap && !queue_empty(pq) ) { // 循环队列
        pq->front = 0;
    }
    pq->size--;
    return pq->arr[pq->front++];
}

    //获取size，存入数据大小
    int queue_size(queue_t* pq){
        return pq->size;
    }
}
